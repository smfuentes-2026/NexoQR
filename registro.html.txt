    <script>
        lucide.createIcons();
        const apiURL = "https://script.google.com/macros/s/AKfycbyMVMlpzn7dG6C_s2DEKz7WsD4-Y7kprvU6FWFm-WxCChWk0xKHMld3QTyZcRsR44nn/exec";
        
        document.getElementById('form-registro').addEventListener('submit', async function(e) {
            e.preventDefault();
            const btn = document.getElementById('btn-submit');
            btn.innerText = "PROCESANDO...";
            btn.disabled = true;

            const datos = {
                nombre: document.getElementById('reg-nombre').value,
                descripcion: document.getElementById('reg-descripcion').value,
                telefono: document.getElementById('reg-telefono').value.replace(/\s+/g, ''),
                linkedin: document.getElementById('reg-linkedin').value,
                avatarOrigen: document.querySelector('input[name="origen-avatar"]:checked').value,
                web1: document.getElementById('reg-web1').value,
                web2: document.getElementById('reg-web2').value
            };

            try {
                await fetch(apiURL, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(datos) });
                // Redirección corregida al archivo raíz index con parámetros limpios
                window.location.href = `./?id=${datos.telefono}&showQR=true`;
            } catch (error) {
                alert("Error al conectar con la base de datos.");
                btn.innerText = "GENERAR MI CONECTAQR";
                btn.disabled = false;
            }
        });

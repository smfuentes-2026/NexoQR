    <script>
        const apiURL = "https://script.google.com/macros/s/AKfycbyMVMlpzn7dG6C_s2DEKz7WsD4-Y7kprvU6FWFm-WxCChWk0xKHMld3QTyZcRsR44nn/exec";
        
        async function cargarPerfil() {
            const urlParams = new URLSearchParams(window.location.search);
            const userId = urlParams.get('id');
            const showQR = urlParams.get('showQR');

            if (!userId) {
                // Si alguien entra sin ID, lo mandamos al registro de forma automática
                window.location.href = "registro.html";
                return;
            }

            try {
                const response = await fetch(`${apiURL}?id=${userId}`);
                const usuario = await response.json();

                if (usuario.error) {
                    document.body.innerHTML = `<div class="text-center text-red-400 font-bold">${usuario.error}</div>`;
                    return;
                }

                // Lógica de Redirección Automática si SÓLO hay teléfono ingresado
                const hasPhone = !!usuario.telefono;
                const hasEmail = !!usuario.email;
                const hasLinkedin = !!usuario.linkedin;
                const hasWeb1 = !!usuario.web1;
                const hasWeb2 = !!usuario.web2;

                const txtWA = encodeURIComponent("Hola, escaneé tu NEXOQR. Me gustaría guardar tu contacto.");
                const waLink = `https://wa.me/${usuario.telefono}?text=${txtWA}`;

                if (hasPhone && !hasEmail && !hasLinkedin && !hasWeb1 && !hasWeb2 && !showQR) {
                    window.location.href = waLink;
                    return;
                }

                // Renderizar datos en la Landing
                document.getElementById('user-name').innerText = usuario.nombre;
                document.getElementById('user-desc').innerText = usuario.descripcion;
                document.getElementById('btn-whatsapp').href = waLink;

                if (hasLinkedin) {
                    document.getElementById('btn-linkedin').href = usuario.linkedin;
                    document.getElementById('btn-linkedin').classList.remove('hidden');
                }
                if (hasEmail) {
                    document.getElementById('btn-email').href = `mailto:${usuario.email}`;
                    document.getElementById('btn-email').classList.remove('hidden');
                }
                if (hasWeb1) {
                    document.getElementById('btn-web1').href = usuario.web1;
                    document.getElementById('btn-web1').classList.remove('hidden');
                }
                if (hasWeb2) {
                    document.getElementById('btn-web2').href = usuario.web2;
                    document.getElementById('btn-web2').classList.remove('hidden');
                }

                // Lógica del nuevo requisito: Mostrar sección de QR si corresponde
                if (showQR === "true") {
                    document.getElementById('qr-section').classList.remove('hidden');
                    // Generamos el código QR dinámico apuntando a esta misma URL de perfil
                    const urlPerfilPublico = window.location.href.split('&')[0]; // Limpiamos el string del parámetro showQR
                    new QRCode(document.getElementById("qrcode"), {
                        text: urlPerfilPublico,
                        width: 180,
                        height: 180,
                        colorDark : "#0f172a",
                        colorLight : "#ffffff"
                    });
                }

                document.getElementById('main-card').classList.remove('hidden');
                lucide.createIcons();

            } catch (error) {
                document.body.innerHTML = `<div class="text-center text-red-400 font-bold">Error de conexión con el servidor de datos.</div>`;
            }
        }

        window.onload = cargarPerfil;
    </script>
